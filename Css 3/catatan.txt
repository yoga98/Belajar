font-family => mengatur jenis font yang akan digunakan
font-size => mengatur ukuran font
font-weight => mengatur ketebalan font
font-variant =>mengubah font menjadi small caps
line-height => mengatur jarak antar baris atau kata atas bawah
color => memberi warna pada tulisan
text-align => mengatur format paragraft/teks
text-indent => memberi indentasi pada paragraf/ teks
text-decoration =>mengatur dekorasi pada teks
text-transform => mengubah jenis huruf menjadi huruf besar,kecil/kapital
letter-spacing => mengatur spasi antar huruf
word-spacing => mengatur spasi antar kata
background-color => mengatur warna pada background
background-image => mengatur gambar yang akan digunakan pada background atau mengatur linear-gradien dan radial-gradien
background-position => mengatur posisi gambar pada background
background-repeat => mengatur jenis pengulanan gambar pada background
pseudo-class => berhubungan dengan link
		-> :link = style default pada sebuah link (a yang memiliki href)
		-> :hover = style ketika kursor mouse berada diatas sebuah link/elemen
		-> :active = style ketika sebuah link di-klik (keadaan aktif)
		->visited = style ketika sebuah link sudah pernah dikunjungi sebelumnya (menggunakan browser sama)

span display default=> inline (tidak bisa di atur tinggi dan lebar)
margin=> area tranparan di sekitar kotak (diluar border)
padding => area transparan di dalam kotak (antara content dan border)
border => batas disekliling content dan padding 
content => konten sebenarnya di dalam box, bisa berupa teks atau gambar
 box-sizing=> mengatur agar lebarnya menjadi tetap tanpa ada perubahan dimensi pada kotak

text-wrapping = membuat texts mengelilingi gambar/element lain

multi-column layout => membuat halamam memiliki beberapa kolom

position deafultnya static => menggeser objek tampa memperngarui lingkup objek di sekitarnya
z-index  => sumbu 3Demensi semakin besar angka yang dimasukan akan semakin dekat dengan kita, atau lebih depan
opacity=> mengatur transparansi sebuah objek
